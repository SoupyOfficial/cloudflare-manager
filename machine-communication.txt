## STEP 2 — Delete the wrong DNS record (mac.madebysoupy.dev)
#
# The first sync run created the wrong subdomain. Delete it via the Cloudflare API:

$env:CF_API_TOKEN = "<your-api-token>"   # set this from your .env or secrets
$env:CF_ZONE_ID   = "<your-zone-id>"

# Find the wrong record ID and delete it:
$headers = @{ Authorization = "Bearer $env:CF_API_TOKEN" }
$records = Invoke-RestMethod "https://api.cloudflare.com/client/v4/zones/$env:CF_ZONE_ID/dns_records?name=mac.madebysoupy.dev" -Headers $headers
$records.result | ForEach-Object {
    Write-Host "Deleting: $($_.name) ($($_.id))"
    Invoke-RestMethod "https://api.cloudflare.com/client/v4/zones/$env:CF_ZONE_ID/dns_records/$($_.id)" -Method Delete -Headers $headers
}
# If no records found that is fine — nothing to delete.
